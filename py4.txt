print("Для подсчета корней введите данные с учётом знаков")
a = float(input("Введите старший коэффицент: "))
b = float(input("Введите коэффицент при первой степени: "))
c = float(input("Введите свободный член: "))
D = (b**2-4*a*c)**0.5
x1 = (-b + D) / (2 * a)
x2 = (-b - D) / (2 * a)
print("Первый корень: ", x1)
print("Второй корень: ", x2)
